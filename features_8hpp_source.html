86 void get_indices_map(std::multimap<FeatId, FeatId>& out, int instance=-1) const
87 {
88 FeatId begin = (instance==1) * static_cast<FeatId>(num_features());
89 FeatId end = (static_cast<FeatId>(instance!=0) + 1) * static_cast<FeatId>(num_features());
90
91 for (FeatId index = begin; index < end; ++index)
92 {
93 FeatId feat_id = get_feat_id(index);
94 out.insert({feat_id, index});
95 }

98 std::multimap<FeatId, FeatId> get_indices_map(int instance=-1) const

105 void share_all_features_between_instances()
106 {
107 for (FeatId index = 0; static_cast<size_t>(index) < names_.size(); ++index)
108 uf_union(index, index+num_features());
109 }

118 AddTree transform(const AddTree& at, int instance=0) const
119 {
120 instance = clean_instance(instance);
121 AddTree new_at;
122 for (const Tree& t : at)
123 {
124 Tree& new_t = new_at.add_tree();

175 // https://en.wikipedia.org/wiki/Disjoint-set_data_structure
176 FeatId uf_find(FeatId index) const
177 {
178 while (feat_ids_[index] != index)
179 {
180 feat_ids_[index] = feat_ids_[feat_ids_[index]];
181 index = feat_ids_[index];
182 }
183 return index;
184 }
185
186 void uf_union(FeatId index1, FeatId index2) {
187 index1 = uf_find(index1);
188 index2 = uf_find(index2);
189
190 if (index1 == index2)
191 return;
192 if (index1 > index2)
193 std::swap(index1, index2);
194
195 feat_ids_[index2] = index1;
196 }

198 int clean_instance(int instance) const { return std::min(1, std::max(0, instance)); }

200 void init()
201 {
202 for (const std::string& name : names_)
203 {
204 index_map_.insert({name, feat_ids_.size()});
205 feat_ids_.push_back(feat_ids_.size());
206 }
207
208 // add len(names) to feat_ids_ for second instance
209 for (size_t i = 0; i < names_.size(); ++i)
210 feat_ids_.push_back(feat_ids_.size());
211
212 }

220 void transform(Tree::ConstRef n, Tree::MutRef m, int instance) const
221 {
222 if (n.is_internal())
223 {
224 LtSplit s = n.get_split();
225 FeatId index = get_index(s.feat_id, instance);
226 if (static_cast<size_t>(index) >= feat_ids_.size())
227 throw std::runtime_error("feature index out of bounds");
228 FeatId feat_id = get_feat_id(index);
229 m.split({feat_id, s.split_value});
230 transform(n.right(), m.right(), instance);
231 transform(n.left(), m.left(), instance);
232 }
233 else
234 {
235 m.set_leaf_value(n.leaf_value());
236 }
237 }
238 };
