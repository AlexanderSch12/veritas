30 template <typename Search, typename State>
31 FloatT compute_basic_output_heuristic_(const Search& s, const State& state) const
32 {
33 FloatT h = 0.0;
34 s.workspace_.leafiter2.setup_flatbox(state.box); // do once
35 for (size_t tree_index = state.indep_set + 1;
36 tree_index < s.at_.size(); ++tree_index)
37 {
38 FloatT max = -FLOATT_INF;
39 const Tree& t = s.at_[tree_index];
40 s.workspace_.leafiter2.setup_tree(t);
41 NodeId leaf_id = -1;
42 while ((leaf_id = s.workspace_.leafiter2.next()) != -1)
43 {
44 if (s.node_box_[tree_index][leaf_id].is_null_box())
45 {
46 //std::cout << "skipping2 " << leaf_id << " because constraints" << std::endl;
47 continue;
48 }
49 max = std::max(t[leaf_id].leaf_value(), max);
50 }
51 h += max;
52 }
53 return h;
54 }

256 template <int p>
257 FloatT compute_lp_heuristic_(
258 const Search<MinDistToExampleHeuristic>& s,
259 const MinDistToExampleState& state) const
260 {
261 FloatT lp_h = 0.0;
262 FloatT lp_state = 0.0;
263 for (auto &&[feat_id, dom] : state.box)
264 {
265 FloatT x = example[feat_id];
266 FloatT d = std::max({dom.lo - x, x - dom.hi, FloatT(0.0)});
267 lp_state += std::pow(d, p);
268 }
269
270 s.workspace_.leafiter2.setup_flatbox(state.box); // do once
271 for (size_t tree_index = state.indep_set + 1;
272 tree_index < s.at_.size(); ++tree_index)
273 {
274 FloatT min_lp = FLOATT_INF;
275 const Tree& t = s.at_[tree_index];
276 s.workspace_.leafiter2.setup_tree(t);
277 NodeId leaf_id = -1;
278 while ((leaf_id = s.workspace_.leafiter2.next()) != -1)
279 {
280 BoxRef box = s.node_box_[tree_index][leaf_id];
281 if (box.is_null_box())
282 continue;
283
284 FloatT lp = lp_state;
285 for (auto &&[feat_id, dom] : box)
286 {
287 Domain dom0;
288 if (static_cast<size_t>(feat_id) <
289 s.workspace_.leafiter2.flatbox.size())
290 dom0 = s.workspace_.leafiter2.flatbox[feat_id];
291 Domain dom1 = dom.intersect(dom0);
292 FloatT x = example[feat_id];
293 FloatT d0 = std::max({dom0.lo - x, x - dom0.hi, FloatT(0.0)});
294 FloatT d1 = std::max({dom1.lo - x, x - dom1.hi, FloatT(0.0)});
295 lp = lp - std::pow(d0, p) + std::pow(d1, p);
296 }
297 min_lp = std::min(min_lp, lp); // pick the leaf with the lowest lp-distance
298 }
299
300 lp_h = std::max(min_lp, lp_h);
301 }
302 return lp_h;
303 }
