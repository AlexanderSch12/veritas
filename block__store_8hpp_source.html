20  template <typename T>
21  class BlockStore {
22  using Block = std::vector<T>;
23  std::vector<Block> blocks_;
24 
25  Block& get_block_with_remaining_capacity(size_t cap, size_t rem_memory_capacity)
26  {
27  Block& block = blocks_.back();
28  if (block.capacity() - block.size() < cap) // allocate a new block
29  {
30  size_t rem_capacity = rem_memory_capacity / sizeof(T);
31  if (rem_capacity > block.capacity() * 2) // double size of blocks each time,..
32  rem_capacity = block.capacity() * 2; // .. unless memory limit almost reached
33  else if (rem_capacity > 0)
34  std::cerr << "WARNING: almost running out of memory, "
35  << static_cast<double>(rem_memory_capacity) / (1024.0*1024.0)
36  << " mb left " << std::endl;
37  else
38  throw std::runtime_error("BlockStore: out of memory");
39 
40  Block new_block;
41  new_block.reserve(rem_capacity);
42  blocks_.push_back(std::move(new_block));
43 
44  std::cout << "BlockStore memory: " << get_mem_size() << " bytes, "
45  << (static_cast<double>(get_mem_size()) / (1024.0 * 1024.0)) << " mb ("
46  << blocks_.size() << " blocks)" << std::endl;
47  }
48 
49  return blocks_.back();
50  }

68  // disallow: references change wrt other blockstore, most likely this is a mistake
69  BlockStore(const BlockStore&) = delete;

74  size_t get_mem_size() const
75  {
76  size_t mem = 0;
77  for (const Block& b : blocks_)
78  mem += b.capacity() * sizeof(T);
79  return mem;
80  }

82  size_t get_used_mem_size() const
83  {
84  size_t mem = 0;
85  for (const Block& b : blocks_)
86  mem += b.size() * sizeof(T);
87  return mem;

90  template <typename IT>
91  Ref store(IT begin, IT end, size_t rem_memory_capacity)
92  {
93  // this store_ block has enough space to accomodate the workspace DomainBox
94  size_t size = end - begin;
95  Block& block = get_block_with_remaining_capacity(size, rem_memory_capacity);
96 
97  // push a copy of the workspace DomainBox
98  size_t start_index = block.size();
99  for (; begin != end; ++begin)
100  block.push_back(*begin);
101 
102  const T *ptr = &block[start_index];
103 
104  return { ptr, ptr + size };
105  }
106 
107  template <typename Container>
108  Ref store(const Container& c, size_t rem_memory_capacity)
109  { return store(c.begin(), c.end(), rem_memory_capacity); }
110  };
