181 inline std::enable_if_t<T::is_mut_type::value, void>
182 split(LtSplit split)
183 {
184 if (is_internal()) throw std::runtime_error("split internal");
185
186 NodeId left_id = tree_->nodes_.size();
187
188 inner::Node left(left_id, id());
189 inner::Node right(left_id + 1, id());
190
191 tree_->nodes_.push_back(left);
192 tree_->nodes_.push_back(right);
193
194 node().internal.split = split;
195 node().internal.left = left_id;
196
197 node().tree_size = 3;
198 NodeRef n(*this);
199 while (!n.is_root())
200 {
201 n = n.parent();
202 n.node().tree_size += 2;
203 }
204 }
