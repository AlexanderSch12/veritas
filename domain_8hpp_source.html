170 inline bool refine_box(Box& box, FeatId feat_id, const Domain& dom)
171 {
172 Domain new_dom;
173 auto it = std::find_if(box.begin(), box.end(), [feat_id](const DomainPair& p)
174 { return p.feat_id == feat_id; });
175
176 if (it != box.end())
177 new_dom = it->domain;
178 if (!new_dom.overlaps(dom))
179 return false;
180
181 new_dom = new_dom.intersect(dom);
182
183 // ensure sorted
184 if (it == box.end())
185 {
186 box.push_back({ feat_id, new_dom });
187 for (int i = box.size() - 1; i > 0; --i) // ids sorted
188 if (box[i-1].feat_id > box[i].feat_id)
189 std::swap(box[i-1], box[i]);
190 //std::sort(workspace_.begin(), workspace_.end(),
191 // [](const DomainPair& a, const DomainPair& b) {
192 // return a.first < b.first;
193 // })
194 }
195 else
196 {
197 it->domain = new_dom;
198 }
199 return true;
200 }

246 inline bool overlaps(const BoxRef& other) const
247 {
248 auto it0 = begin_;
249 auto it1 = other.begin_;
250
251 while (it0 != end_ && it1 != other.end_)
252 {
253 if (it0->feat_id == it1->feat_id)
254 {
255 if (!it0->domain.overlaps(it1->domain))
256 return false;
257 ++it0; ++it1;
258 }
259 else if (it0->feat_id < it1->feat_id) ++it0;

261 }

291 inline int overlaps(const LtSplit& split) const
292 {
293 for (auto it = begin_; it < end_; ++it)
294 {
295 if (it->feat_id == split.feat_id)
296 {

334 inline void
335 combine_boxes(const BoxRef& a, const BoxRef& b, bool copy_b, Box& out)
336 {
337 if (!out.empty())
338 throw std::runtime_error("output box is not empty");
339
340 const DomainPair *it0 = a.begin();
341 const DomainPair *it1 = b.begin();
342
343 // assume sorted
344 while (it0 != a.end() && it1 != b.end())
345 {
346 if (it0->feat_id == it1->feat_id)
347 {
348 Domain dom = it0->domain.intersect(it1->domain);
349 out.push_back({ it0->feat_id, dom });
350 ++it0; ++it1;
351 }
352 else if (it0->feat_id < it1->feat_id)
353 {
354 out.push_back(*it0); // copy
355 ++it0;
356 }
357 else
358 {
359 if (copy_b)
360 out.push_back(*it1); // copy
361 ++it1;
362 }
363 }
364
365 // push all remaining items (one of them is already at the end, no need to compare anymore)
366 for (; it0 != a.end(); ++it0)
367 out.push_back(*it0); // copy
368 for (; copy_b && it1 != b.end(); ++it1)
369 out.push_back(*it1); // copy
370 }
